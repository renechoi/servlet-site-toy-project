-- WEBDB.USERS definition
DROP TABLE USERS;

CREATE TABLE USERS
(
    NO       NUMBER,
    NAME     VARCHAR2(80),
    EMAIL    VARCHAR2(100) NOT NULL,
    PASSWORD VARCHAR2(100) NOT NULL,
    GENDER   VARCHAR2(10),
    PRIMARY KEY (NO)
);

-- WEBDB.BOARD definition
DROP TABLE BOARD;

CREATE TABLE BOARD
(
    NO       NUMBER,
    TITLE    VARCHAR2(500),
    CONTENT  VARCHAR2(4000),
    HIT      NUMBER,
    REG_DATE DATE,
    USER_NO  NUMBER,
    PRIMARY KEY (NO),
        CONSTRAINT C_BOARD_FK FOREIGN KEY (USER_NO)
        REFERENCES USERS (NO)
);


--시퀀스 객체(USERS 용)
DROP SEQUENCE SEQ_USERS_NO;
CREATE SEQUENCE SEQ_USERS_NO
    INCREMENT BY 1
    START WITH 1;

--시퀀스 객체(BOARD 용)
DROP SEQUENCE SEQ_BOARD_NO;
CREATE SEQUENCE SEQ_BOARD_NO
    INCREMENT BY 1
    START WITH 1;
